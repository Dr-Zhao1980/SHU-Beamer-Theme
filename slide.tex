% 第二页：围绕导师方向拟系统学习的三个模块
\begin{frame}[t, shrink=5]{当前科研学习内容（一）}
\small

\begin{block}{双曲守恒律}
当前主要学习双曲守恒律的基本形式
\[
U_t+F(U)_x=0,
\]
并理解其弱解、熵条件与基本波结构。

\vspace{0.3em}
{\scriptsize 参考文献：\cite{Dafermos2016,Bressan2000,Lax1957,Glimm1965}.}
\end{block}

\begin{block}{黎曼问题}
重点关注分段常数初值
\[
U(x,0)=
\begin{cases}
U_L, & x<0,\\
U_R, & x>0,
\end{cases}
\]
以及激波、稀疏波、接触间断等典型波现象。

\vspace{0.3em}
{\scriptsize 参考文献：\cite{ChangHsiao1989,Toro2009,ZhangZheng1990}.}
\end{block}

\end{frame}





\begin{frame}[t, shrink=5]{当前科研学习内容（二）}
\small

\begin{block}{可压 Euler 方程组}
当前重点学习一维可压 Euler 方程组
\[
\begin{cases}
\rho_t+(\rho u)_x=0,\\
(\rho u)_t+(\rho u^2+p)_x=0,\\
E_t+\bigl(u(E+p)\bigr)_x=0,
\end{cases}
\]
并初步理解其基本波结构。

\vspace{0.3em}
{\scriptsize 参考文献：\cite{ChangHsiao1989,Toro2009}.}
\end{block}

\begin{block}{进一步关注的问题}
在此基础上，进一步关注二维 Riemann 问题中的波相互作用、
跨声激波与自由边界问题，以及含反应项模型中的波传播问题。

\vspace{0.3em}
{\scriptsize 参考文献：\cite{ZhangZheng1990,ChangChenYang1995,ChangChenYang2000,ChenFeldman2003,ChenFeldman2018,Smoller1994}.}
\end{block}

\end{frame}

\section{研究方向理解与问题认识}
% ============================================================================

\begin{frame}[shrink=5]{选择该方向的原因}
\begin{block}{为什么选择黎曼问题作为切入点？}
黎曼问题是双曲守恒律研究中的基本模型，它能够集中体现激波、稀疏波、接触间断等典型波结构的形成机制，是理解更复杂初边值问题的重要起点。
\end{block}

\begin{itemize}
    \item \textbf{从一维到高维：}高维问题中的波相互作用更复杂，几何结构和自由边界现象更加突出；
    \item \textbf{从理想到实际：}真空、反应项、边界效应以及跨声激波与反射--衍射等现象会显著提高问题难度；
    \item \textbf{从定性到定量：}不仅需要知道“会出现什么波”，还需要分析其相互作用机制、存在区间和稳定性特征。
\end{itemize}

\vspace{0.2em}
{\scriptsize 参考文献：\cite{ZhangZheng1990,ChenFeldman2003,ChenFeldman2018}.}

\begin{alertblock}{个人认识}
我希望以黎曼问题为基本切入点，在扎实掌握经典理论的基础上，逐步过渡到高维、自由边界和反应流等更复杂的问题。
\end{alertblock}
\end{frame}
